\subsubsection{Iteratoren}

\begin{frame}{Iteratoren}
    Der Begriff \emph{Iterator} stammt aus dem Bereich der Softwareentwicklung und bezeichnet einen Zeiger, mit dem die Elemente einer Menge durchlaufen werden k√∂nnen (z. B. eine Liste, ein Vector etc.).
\end{frame}

\begin{frame}[fragile]{Iteratoren - Beispiel (Code)}
    \begin{minted}{cpp}
std::vector<int> v{ 3, 1, 4, 1, 5, ... };

// std::vector<int>::iterator
for (auto it = v.begin(); it != v.end(); ++it)
    std::cout << *it << ", ";

std::cout << std::endl;

// 3, 1, 4, 5, 1, ...,
    \end{minted}
\end{frame}
